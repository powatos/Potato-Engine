#Targets
LIB_NAME	:= PotatoEngine
LIB_TARGET	:= lib$(LIB_NAME).a
GAME_TARGET := Game

CXX			:= g++

CXXFLAGS 	:= -g -Wall -std=c++17
LDLIBS   	:= -lncurses

SRCDIR   	:= Source
OBJDIR   	:= Binaries
LOGDIR   	:= logs/debug.log

ENGINE_INCLUDES	:= -I$(SRCDIR)/Internal/Public -I$(SRCDIR)/Internal/Private
GAME_INCLUDES	:= -I$(SRCDIR)/Internal/Public

ENGINE_SOURCES	:= $(shell find $(SRCDIR)/Internal -name '*.cpp')
ENGINE_OBJECTS	:= $(ENGINE_SOURCES:$(SRCDIR)/Internal/%.cpp=$(OBJDIR)/Internal/%.o)

# Create matching .o files in the obj/ folder
GAME_SOURCES	:= $(shell find $(SRCDIR)/Program -name '*.cpp')
GAME_OBJECTS	:= $(GAME_SOURCES:$(SRCDIR)/Program/%.cpp=$(OBJDIR)/Program/%.o)


all: $(GAME_TARGET)

#build engine lib
$(LIB_TARGET): $(ENGINE_OBJECTS)
	@echo "> ARCHIVING ENGINE..."
	ar rcs $@ $^

#build game exe
$(GAME_TARGET): $(LIB_TARGET) $(GAME_OBJECTS)
	@echo "> LINKING GAME..."
	$(CXX) $(CXXFLAGS) $(GAME_INCLUDES) -o $@ $(GAME_OBJECTS) -L. -l$(LIB_NAME) $(LDLIBS)

#engine object rule (both public and private)
$(OBJDIR)/Internal/%.o: $(SRCDIR)/Internal/%.cpp
	@mkdir -p $(dir $@)
	$(CXX) $(CXXFLAGS) $(ENGINE_INCLUDES) -c $< -o $@

#game object rule (only public)
$(OBJDIR)/Program/%.o: $(SRCDIR)/Program/%.cpp
	@mkdir -p $(dir $@)
	$(CXX) $(CXXFLAGS) $(GAME_INCLUDES) -c $< -o $@

flush:
	@> ${LOGDIR}
	@echo "Flushed ${LOGDIR}"

clean:
	@rm -rf $(OBJDIR) $(LIB_TARGET) $(GAME_TARGET)
	@echo "Removed ${OBJDIR}"
	@echo "Removed ${GAME_TARGET}"

	@$(MAKE) --no-print-directory flush

run: all
	@./$(GAME_TARGET)

.PHONY: all clean run flush
